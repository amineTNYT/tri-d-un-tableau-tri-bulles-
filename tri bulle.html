def tri_bulle(t, n):
    permut = True
    while permut == True:
        permut = False
        for i in range(n-1):
            if t[i] > t[i+1]:
                aux = t[i]
                t[i] = t[i+1]
                t[i+1] = aux
                permut = True

# Exemple d'utilisation
t = [5, 3, 8, 1, 4]
tri_bulle(t, 5)
